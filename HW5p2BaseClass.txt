package com.company;

import org.openqa.selenium.By;
import org.openqa.selenium.Keys;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;

public static class BasePage {

    public WebDriverWait driverWait;
    public static String SearchString;
    public static WebDriver driver;
    public static String headerOfFirstResultText;

    public BasePage(WebDriver driver) {
        this.driver = driver;
    }

    public static void click(By locator) {
        WebDriverWait wait = new WebDriverWait(driver,5);
        wait.until(ExpectedConditions.visibilityOfElementLocated(locator);
        driver.findElement(locator).click();

    }
    public static void sendKeys(By locator, String searchString) {
        driver.findElement(locator).sendKeys(SearchString + Keys.RETURN);

    }
    public static String findAndPrintFirstResult(By locator) {
        WebDriverWait wait = new WebDriverWait(driver,5);
        wait.until(ExpectedConditions.visibilityOfElementLocated(locator);
        WebElement firstResultText = driver.findElement(locator);
        headerOfFirstResultText = firstResultText.getText();
        return headerOfFirstResultText;
    }
}
